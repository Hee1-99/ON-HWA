1. Supabase SQL — 테이블 생성

-- ==========================================
-- ON:HWA Database Schema
-- ==========================================

-- UUID 확장 활성화
CREATE EXTENSION IF NOT EXISTS "uuid-ossp";

-- ==========================================
-- 1. shops (플로리스트 가게)
-- ==========================================
CREATE TABLE public.shops (
  id          UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  owner_id    UUID NOT NULL REFERENCES auth.users(id) ON DELETE CASCADE,
  name        TEXT NOT NULL,
  logo_url    TEXT,
  slug        TEXT NOT NULL UNIQUE,
  contact     TEXT,
  plan_type   TEXT NOT NULL DEFAULT 'free' CHECK (plan_type IN ('free', 'pro', 'enterprise')),
  created_at  TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at  TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_shops_owner_id ON public.shops(owner_id);
CREATE INDEX idx_shops_slug ON public.shops(slug);

-- ==========================================
-- 2. bouquets (꽃다발 상품)
-- ==========================================
CREATE TABLE public.bouquets (
  id               UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  shop_id          UUID NOT NULL REFERENCES public.shops(id) ON DELETE CASCADE,
  ai_name          TEXT,
  ai_story         TEXT,
  original_img_url TEXT NOT NULL,
  link_id          TEXT NOT NULL UNIQUE DEFAULT encode(gen_random_bytes(6), 'hex'),
  status           TEXT NOT NULL DEFAULT 'draft' CHECK (status IN ('draft', 'sent', 'archived')),
  recipient_phone  TEXT,
  sent_at          TIMESTAMPTZ,
  created_at       TIMESTAMPTZ NOT NULL DEFAULT NOW(),
  updated_at       TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_bouquets_shop_id ON public.bouquets(shop_id);
CREATE INDEX idx_bouquets_link_id ON public.bouquets(link_id);

-- ==========================================
-- 3. archives (수령인 포토카드 아카이브)
-- ==========================================
CREATE TABLE public.archives (
  id           UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  bouquet_id   UUID NOT NULL REFERENCES public.bouquets(id) ON DELETE CASCADE,
  user_img_url TEXT,
  card_img_url TEXT NOT NULL,
  visitor_id   TEXT,  -- 비로그인 방문자 식별용 (localStorage UUID)
  created_at   TIMESTAMPTZ NOT NULL DEFAULT NOW()
);

CREATE INDEX idx_archives_bouquet_id ON public.archives(bouquet_id);

-- ==========================================
-- updated_at 자동 갱신 트리거
-- ==========================================
CREATE OR REPLACE FUNCTION update_updated_at_column()
RETURNS TRIGGER AS $$
BEGIN
  NEW.updated_at = NOW();
  RETURN NEW;
END;
$$ LANGUAGE plpgsql;

CREATE TRIGGER set_shops_updated_at
  BEFORE UPDATE ON public.shops
  FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();

CREATE TRIGGER set_bouquets_updated_at
  BEFORE UPDATE ON public.bouquets
  FOR EACH ROW EXECUTE FUNCTION update_updated_at_column();
2. RLS (Row Level Security) 정책

-- ==========================================
-- RLS 활성화
-- ==========================================
ALTER TABLE public.shops    ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.bouquets ENABLE ROW LEVEL SECURITY;
ALTER TABLE public.archives ENABLE ROW LEVEL SECURITY;

-- ==========================================
-- shops 정책 (사장님 본인 데이터만)
-- ==========================================
-- 조회: 본인 가게만
CREATE POLICY "shops: owner can select"
  ON public.shops FOR SELECT
  USING (auth.uid() = owner_id);

-- 생성: 로그인 사용자만
CREATE POLICY "shops: owner can insert"
  ON public.shops FOR INSERT
  WITH CHECK (auth.uid() = owner_id);

-- 수정: 본인 가게만
CREATE POLICY "shops: owner can update"
  ON public.shops FOR UPDATE
  USING (auth.uid() = owner_id)
  WITH CHECK (auth.uid() = owner_id);

-- 삭제: 본인 가게만
CREATE POLICY "shops: owner can delete"
  ON public.shops FOR DELETE
  USING (auth.uid() = owner_id);

-- ==========================================
-- bouquets 정책
-- ==========================================
-- 조회: 가게 오너 OR 링크 ID로 접근하는 수령인 (공개)
CREATE POLICY "bouquets: owner can select"
  ON public.bouquets FOR SELECT
  USING (
    shop_id IN (SELECT id FROM public.shops WHERE owner_id = auth.uid())
    OR status IN ('sent', 'archived')  -- 수령인은 발송된 것만 조회 가능
  );

-- 생성/수정/삭제: 본인 가게의 꽃다발만
CREATE POLICY "bouquets: owner can insert"
  ON public.bouquets FOR INSERT
  WITH CHECK (
    shop_id IN (SELECT id FROM public.shops WHERE owner_id = auth.uid())
  );

CREATE POLICY "bouquets: owner can update"
  ON public.bouquets FOR UPDATE
  USING (
    shop_id IN (SELECT id FROM public.shops WHERE owner_id = auth.uid())
  );

CREATE POLICY "bouquets: owner can delete"
  ON public.bouquets FOR DELETE
  USING (
    shop_id IN (SELECT id FROM public.shops WHERE owner_id = auth.uid())
  );

-- ==========================================
-- archives 정책
-- ==========================================
-- 조회: 가게 오너 OR 해당 bouquet의 visitor_id 소유자
CREATE POLICY "archives: public insert (수령인 카드 저장)"
  ON public.archives FOR INSERT
  WITH CHECK (
    bouquet_id IN (
      SELECT id FROM public.bouquets WHERE status IN ('sent', 'archived')
    )
  );

CREATE POLICY "archives: owner can select"
  ON public.archives FOR SELECT
  USING (
    bouquet_id IN (
      SELECT b.id FROM public.bouquets b
      JOIN public.shops s ON b.shop_id = s.id
      WHERE s.owner_id = auth.uid()
    )
  );

-- 수령인 본인 카드 조회 (visitor_id 기반)
CREATE POLICY "archives: visitor can select own cards"
  ON public.archives FOR SELECT
  USING (visitor_id = current_setting('request.headers', true)::json->>'x-visitor-id');
